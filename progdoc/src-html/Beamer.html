001/**
002 * La classe Beamer est une classe decrivant le fonctionnement de l'Item Beamer.
003 * @author Théo GUEURET
004 * @version  05/05/2019
005 */
006public class Beamer extends Item
007{
008    private Room aMemorizedRoom;
009    private boolean aIsCharged;
010    
011    public Beamer(final String pDescr, final int pPoids, final String pName, final boolean pIsEatable, final String pEffect, final int pStrenght, final int pDefense, final int pMiss)
012    {
013        super(pDescr, pPoids, pName, pIsEatable, pEffect, pStrenght, pDefense, pMiss);
014        this.aMemorizedRoom = null;
015        this.aIsCharged = false;
016    }
017    
018    /**
019     * Accesseur de l'attribut aMemorizedRoom.
020     * @return un Room : la Room qui à été mémorisée dans le Beamer.
021     */
022    public Room getMemorizedRoom() {
023        return this.aMemorizedRoom;
024    }
025    
026    /**
027     * Accesseur de l'attribut aIsCharged.
028     * @return un boolean décrivant si le Beamer à été chargé ou non.
029     */
030    public boolean getIsCharged() {
031        return this.aIsCharged;
032    }
033    
034    /**
035     * Modificateur de l'attribut aMemorizedRoom
036     * @param pMemorizedRoom une Room étant celle qui va être mémorisée par le Beamer.
037     */
038    public void setMemorizedRoom(final Room pMemorizedRoom) {
039        this.aMemorizedRoom = pMemorizedRoom;
040    }
041    
042    /**
043     * Modificateur de l'attribut aIsCharged
044     * @param pIsCharged un boolean devenant la nouvelle valeur de l'attribut aIsCharged
045     */
046    public void setIsCharged(final boolean pIsCharged) {
047        this.aIsCharged = pIsCharged;
048    }
049    
050    /**
051     * Fonction représentant l'action de charger un Beamer.
052     * @param pRoom une Room : celle qui va être mémorisée par le Beamer.
053     */
054    public void charge(final Room pRoom) 
055    {
056        setIsCharged(true);
057        setMemorizedRoom(pRoom);
058    }
059    
060    /**
061     * Fonction représentant l'action d'utiliser un Beamer.
062     * @return une Room : la Room dans laquelle le PLayer ayant activé le Beamer va être déplacé.
063     */
064    public Room fire()
065    {
066        setIsCharged(false);
067        return getMemorizedRoom();
068    }
069}
