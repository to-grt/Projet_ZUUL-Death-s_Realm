001import java.util.HashMap;
002import java.util.Set;
003import java.util.Iterator;
004import java.util.ArrayList;
005import java.util.Collection;
006import java.util.Random;
007
008/**
009 * Classe Room, gère les pièces du jeu, leurs data et le traitement de celle-ci
010 * 
011 * @author Théo Gueuret
012 * @version 12/02/2019
013 */
014public class Room
015{
016    private String aDescription;
017    private HashMap<String, Room> aExits;
018    private String aImageName;
019    private ItemList aItems;
020    
021    public Room(final String pDescription, final String pImage)
022    {
023        this.aDescription = pDescription;
024        this.aExits = new HashMap<String, Room>();
025        this.aImageName = pImage;
026        this.aItems = new ItemList();
027    }
028    
029    /**
030     * Ajoute un Item dans l'attribut aItems étant un ItemList. Cette fonction permet donc d'ajouter un Item dans la
031     * Room voulue.
032     * @param pItem un Item valide qui va être ajouté dans la Room voulue en agrandissant l'attriut aItems.
033     */
034    public void addItem(final Item pItem)
035    {
036        aItems.addItem(pItem.getName(), pItem);
037    }
038    
039    /**
040     * Retire un Item dans l'attribut aItems étant un ItemList. Cette fonction permet donc de retirer un Item de la
041     * Room voulue.
042     * @param pItem un Item valide qui va être retiré dans la Room voulue en réduisant l'attribut aItems.
043     */
044    public void removeItem(final Item pItem)
045    {
046        aItems.removeItem(pItem.getName());
047    }
048    
049    /**
050     * Ajoute une sortie à une pièce
051     * @param pDirection un String contenant la direction voulue.
052     * @param pNeighboor une Room, désigne la pièce affectée à la direction
053     */
054    public void setExit(final String pDirection, final Room pNeighboor)
055    {
056        aExits.put(pDirection, pNeighboor);
057    } //setExit()
058    
059    /**
060     * Retourne un String contenant la description de la pièce souhaitée.
061     * 
062     * @return this.aDescription
063     */
064    public String getDescription()
065    {
066        return(this.aDescription + "\n");
067    } //getDescription()
068    
069    /**
070     * Retourne les Item liés à la room.
071     * 
072     * @return une HashMap de String et d'Items des Item situés dans la Room.
073     */
074    public HashMap getItems()
075    {
076        return(this.aItems.getItemList());
077    }
078    
079    /**
080     * Retourne l'Item voulu
081     * @param vName un String contenant le nom d'un Item.
082     * @return un Item tel que son nom est le paramêtre de la fonction.
083     */
084    public Item getItem(final String vName)
085    {
086        return( this.aItems.getItem( vName ) );
087    }
088    
089    /**
090     * Retourne une pièce (Room) située dans la direction voulue par rapport à la pièce actuelle
091     * @param pDirection String indiquant la direction
092     * @param pRooms une Collection de Room : les Room de la salle.
093     * @param pSize un int contenant le nombre de Rooms à selectionner.
094     * @return aExits.get(pDirection)
095     */
096    public Room getExit(final String pDirection, final Collection<Room> pRooms, final int pSize)
097    {
098        if( this.aImageName.equals("Images/Trap.jpg") ) {
099            return(getRandomRoom(pRooms, pSize));
100        }
101        return( aExits.get(pDirection) );
102    } //getExit
103    
104    /**
105     * Retourne une pièce aléatoire.
106     * @param pRooms une Collection de Room : les Rooms pouvant etre selectionnées aléatoirement.
107     * @param pSize un Int contenant la taille de la liste des Room selectionnables. 
108     * @return une Room aléatoire parmis celle disponible(toute sauf la Room finale et elle meme)
109     */
110    public Room getRandomRoom(final Collection<Room> pRooms, final int pSize) {
111        Random randomizer = new Random();
112        Iterator<Room> vListRoom = pRooms.iterator();
113        int vCompteur = 0;
114        int vMax = randomizer.nextInt(pSize);
115        Room vResult = new Room("erreur", "erreur");
116        while( vCompteur <= vMax ) {
117            vResult = vListRoom.next();
118            vCompteur++;
119        }
120        //System.out.println(vResult.
121        if( vResult.getImageName().equals("Final.jpg") || vResult.getImageName().equals("Trap;jpg") ) { return( getRandomRoom(pRooms, pSize) );}
122        return vResult;
123    }
124    
125    /**
126     * Retourne un String indiquant toute les sortie possible d'une pièce spécifique
127     * @return un String contenant les directions de sorties existantes.
128     */
129    public String getExitString()
130    {
131        String vString = new String("Exits :");
132        Set<String> keys = aExits.keySet();
133        for( String exit : keys)
134        {
135            vString += " " + exit;
136        }
137        return( vString );
138    } //getExitString()
139
140    /**
141     * Retourne une description des items de la salle
142     * @return un String contenant la description formatée des Item de la Room.
143     */
144    public String getItemsDescr()
145    {
146        String vString = new String();
147        for(Item vItem : this.aItems.getValues()) vString += "this " + vItem.getItemDescr() + "\n";
148        return(vString);
149    }
150        
151    /**
152     * Retourne une longue description de la pièce voulue.
153     * @return un String une description formatée de la Room.
154     */
155    public String getLongDescription()
156    {
157        return("You are " + aDescription + ".\n" + getExitString() + "\n");
158    } //getLongDescription()
159
160    /**
161     * Retourne un String decrivant le nom de l'image de la piece
162     * @return un String contenant le nom de l'image adpté à la pièce.
163     */
164    public String getImageName()
165    {
166        return(this.aImageName);
167    }
168    
169    /**
170     * Retourne un boolean permettant de savoir si la pièce passée en paramètre fait partie des sorties de la Room actuelle
171     * @param pRoom, une Room, on veut savoir si cettte Room fait partie des sorties de la salle actuelle.
172     * @return un boolean permettant de savoir si cette Room est une exit de la salle actuelle ou non .
173     */
174    public boolean isExit(final Room pRoom)
175    {
176        Collection<Room> vRooms = this.aExits.values();
177        for( Room vRoom : vRooms ) {
178            if( vRoom == pRoom ) {
179                return true;
180            }
181        }
182        return false;
183    }
184} // Room
