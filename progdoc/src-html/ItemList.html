001import java.util.Collection;
002import java.util.HashMap;
003
004/** 
005 * Classe permettant de gerer une collection d'item
006 *
007 * @author Théo Gueuret
008 * @version 10/04/2019
009 */
010public class ItemList
011{
012    private HashMap<String, Item> aItemList;
013
014    /**
015     * Constructeur de la classe ItemList.
016     */
017    public ItemList()
018    {
019        this.aItemList = new HashMap<String, Item>();
020    }
021    
022    /**
023     * Accesseur de l'attribut aItemList de la classe ItemList.
024     *  @return un HashMap contenant les Items de l'ItemList.
025     */
026    public HashMap getItemList() 
027    {
028        return( this.aItemList );
029    }
030    
031    /**
032     * Fonction permettant de retourner un Item précis.
033     * @param vName un String contenant le nom d'un Item existant.
034     * @return un Item tel que le nom de celui-ci est le String passé en paramètre.
035     */
036    public Item getItem(final String vName) 
037    {
038        return( this.aItemList.get( vName ) );
039    }
040    
041    /**
042     * Modificateur de l'attribut aItemList de la classe ItemList.
043     * @param pItemList une HashMap de String et d'Items.
044     */
045    public void setItemList(final HashMap pItemList)
046    {
047        this.aItemList = pItemList;
048    }
049    
050    /**
051     * Fonction permettant d'ajouter un Item dans une ItemList
052     * @param pItemName un String contenant le nom d'un Item.
053     * @param pItem     un Item, son nom est le même que le celui contenu dans le String du précedent paramètre.
054     */
055    public void addItem(final String pItemName, final Item pItem)
056    {
057        this.aItemList.put(pItemName, pItem);
058    }
059
060    /**
061     * Fonction permettant de retirer un Item d'une ItemList
062     * @param pItemName un String contenant le nom de l'Item voulant être retiré
063     */
064    public void removeItem(final String pItemName)
065    {
066        this.aItemList.remove(pItemName);
067    }
068    
069    /**
070     * Fonction permettant de retourner une collection de toutes les values de l'attribut aItemList.
071     * @return une Collecion d'Item correspondant a toutes les values de l'attribut aItemList. Ceci est possible 
072     * grâce au fait que aItemList est une HashMap et que certaines fonctions sont applicables sur celle-ci dont .values().
073     */
074    public Collection<Item> getValues()
075    {
076        return( this.aItemList.values() );
077    }
078    
079    /**
080     * Fonction permettant de récuperer la liste des items de la ItemList ainsi que leur poids sous forme de String
081     * @return un String contenant la liste des Item contenus dans ItemList (via l'attribut aItemList).
082     */
083    public String getTotalString()
084    {
085        String vTotal = "";
086        Collection<Item> vItems = this.aItemList.values();
087        for( Item vItem : vItems ) {
088            vTotal += "a " + vItem.getItemDescr() + "\n";
089        }
090        return( vTotal );
091    }
092    
093    /**
094     * Fonction retournant la somme du poids de tous les items de la ItemList.
095     * @return un int symbolisant la somme des poids de tous les Items de la ItemList.
096     */
097    public int getTotalWeight()
098    {
099        int vResult = 0;
100        Collection<Item> vItems = this.aItemList.values();
101        for( Item vItem : vItems) {
102            vResult += vItem.getPoids();
103        }
104        return( vResult );
105    }
106}
