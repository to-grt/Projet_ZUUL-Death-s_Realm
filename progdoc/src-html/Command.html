001/**
002 * Classe Command, cette classe gère les informations des objets de type Command, elle permet d'acceder aux attibut des instances
003 * de la classe mais elle sert aussi à savoir si c'est une commande valide ou encore si elle comporte 2 mots.
004 * 
005 * @author Théo Gueuret
006 * @version 12/02/2019
007 */
008public class Command
009{
010    private String aCommandWord;
011    private String aSecondWord;
012    
013    /**
014     * Constructeur de la classe Command
015     * @param pCommandWord première partie de la commande.
016     * @param pSecondWord  deuxième partie de la commande.
017     */
018    public Command(final String pCommandWord, final String pSecondWord)
019    {
020        this.aCommandWord = pCommandWord;
021        this.aSecondWord = pSecondWord;
022    }
023    
024    /**
025     * accesseur de l'attribut aCommandWord
026     * @return un String contenant le premier mot de l'objte Command voulue.
027     */
028    public String getCommandWord()
029    {
030        return(this.aCommandWord);
031    }
032    
033    /**
034     * accesseur de l'attribut aSecondWord.
035     * @return un String contenant le deuxième mot de l'objet Command voulue.
036     */
037    public String getSecondWord()
038    {
039        return(this.aSecondWord);
040    }
041    
042    /**
043     * Fonction permettant de savoir si il y a présence ou non d'un second mot dans la commande.
044     * @return un booolean permettant de savoir si le deuxième mot de la Command est nul ou non.
045     */
046    public boolean hasSecondWord()
047    {
048        return(this.aSecondWord != null);
049    }
050    
051    /**
052     * Fonction permettant de savoir si la commande est connue ou non.
053     * @return un boolean permettant de savoir si la Command est connue ou non.
054     */
055    public boolean isUnknown()
056    {
057        return(this.aCommandWord == null);
058    }
059} // Command
