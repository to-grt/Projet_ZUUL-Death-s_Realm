170    /**
171     * Execute la fonction adaptée à la commande passée en paramètre, si la commande n'existe pas un message d'erreur est affiché
172     * @param pCommandLine un String contenant la commande entrée par l'utilisateur.
173     */
174    public void interpretCommand(String pCommandLine) 
175    {
176        this.aGui.println("");
177        this.aGui.println(pCommandLine);
178        Command vCommand = this.aParser.getCommand(pCommandLine);
179
180        if(vCommand.isUnknown()) {
181            this.aGui.println("I don't know what you mean...");
182            return;
183        }
184        String vCommandWord = vCommand.getCommandWord();
185        if (vCommandWord.equals("help")) {
186            printHelp();
187        }
188        else if (vCommandWord.equals("go")) {
189            goRoom(vCommand);
190        }
191        else if (vCommandWord.equals("quit")) {
192            if(vCommand.hasSecondWord()) {
193                this.aGui.println("Quit what?" );
194                return;
195            }
196            else {
197                endGame();
198            }
199        }
200        else if(vCommandWord.equals("look")) {
201            look(vCommand);
202        }
203        else if(vCommandWord.equals("eat")) {
204            eat(vCommand);
205        }
206        else if(vCommandWord.equals("back")) {
207            back(vCommand);
208        }
209        else if(vCommandWord.equals("test")) {
210            if( !vCommand.hasSecondWord() ) {
211                this.aGui.println("test what ?");
212                return;
213            }
214            test( vCommand.getSecondWord() );
215        }
216        else if(vCommandWord.equals("take")) {
217            take(vCommand);
218        }
219        else if(vCommandWord.equals("drop")) {
220            drop(vCommand);
221        }
222        else if(vCommandWord.equals("inventory")) {
223            inventory(vCommand);
224        }
225        else if(vCommandWord.equals("charge")) {
226            charge(vCommand);
227        }
228        else if(vCommandWord.equals("fire")) {
229            fire(vCommand);
230        }
231        else {
232            endGame();
233            return;
234        }
235        this.aTimeLeft--;
236        if( this.aTimeLeft == 25 ) {
237            this.aGui.println("The death's aura of the temple is biting you gradually, you must take care !");
238        }
239        if( this.aTimeLeft == 0 ) {
240            this.aGui.println("This was your last move \nThe death's aura of the Temple managed to slay you down, you are dead.");
241            endGame();        
242            return;
243        }
244    }

246    /**
247     * Permet de lire les commandes d'un fichier texte.
248     * @param pString un String contenant le nom du fichier à executé sans le '.txt' .
249     */
250    public void test(String pString) {
251        Scanner vSc;
252        try {
253            vSc = new Scanner ( new File( "essais/" + pString + ".txt" ) );
254            while ( vSc.hasNextLine() ) {
255                String vCommand = vSc.nextLine();
256                interpretCommand(vCommand);
257            }
258        } //try
259        catch( final FileNotFoundException pE ) {
260            this.aGui.println(" This file doesn't exist ! \n");
261            return;
262        }
263        
264    }

268     * @return un String contenant les Items de la salle.
269     */
270    public String getItemsInfo()
271    {
272        String vResult = "";
273        vResult = ( "In this room you can find : \n");
274        if( !this.aPlayer.getRoomItems().isEmpty() ) {
275            Collection<Item> vItems = this.aPlayer.getRoomItems().values();
276            for(Item vItem : vItems) {
277                vResult += ( "  a " + vItem.getName() );
278            }
279            vResult += (".");
280        }
281        else vResult += ("nothing \n");
282        return vResult;
283    }

316    /**
317     * Reviens à la salle precedente
318     * @param pCommand une Command permettant de vérifiée si les conditions pour revenir en arrière sont bien respectées.
319     */
320    private void back(Command pCommand)
321    {
322        if( pCommand.hasSecondWord() ) {
323            this.aGui.println("You can't do that ! 'back' is a single word command \n");
324            return;
325        }
326        if( this.aPlayer.getPreviousRoom().empty() ) {
327            this.aTimeLeft++;
328            this.aGui.println("Vous etes deja revenu au debut de votre aventure \n");
329            return;
330        }
331        Room vRoom = this.aPlayer.removeRoom();
332        if( this.aPlayer.getCurrentRoom().isExit(vRoom)) {
333            this.aPlayer.changeRoom( vRoom );
334            printLocationInfo();
335            if(this.aPlayer.getRoomImageName() != null) this.aGui.showImage(this.aPlayer.getRoomImageName());
336        }
337        else {
338            this.aGui.println("It was a trapped door ! \n Sorry you can't go back there !");
339            this.aPlayer.addRoom(vRoom);
340        }
341        
342    }

402    /**
403     * Fonction permettant de prendre un item d'une salle si cela est possible
404     * @param pCommand une Command permettant de vérifier si toutes les actions sont vérifiées avant de prendre un Item de la Room du joueur.
405     */
406    public void take(final Command pCommand) {
407        if(!pCommand.hasSecondWord()) 
408        {
409            this.aGui.println("Take what ? \n");
410            return;
411        }
412        
413        String vWantedItem = pCommand.getSecondWord();
414        
415        if(this.aPlayer.getItemFromRoom( vWantedItem ) != null)
416        {
417            Item vItem = (Item)(this.aPlayer.getItemFromRoom( vWantedItem ));
418            if( (vItem.getPoids() + this.aPlayer.getCurrentWeight()) <= this.aPlayer.getMaxWeight())
419            {
420                this.aGui.println( this.aPlayer.pickItem(vItem) );
421                this.aPlayer.getCurrentRoom().removeItem(vItem);
422            }
423            else this.aGui.println("Too heavy for you ! You can't pick that up. \n");
424        }
425        else this.aGui.println("This item : '" + vWantedItem + "' does not exist or is not in this room. \n");
426        this.aGui.update();
427    }

479    /**
480     * Selecteur qui permet de diriger le joueur selon la commande entrée par l'utilisateur.
481     * @param pCommand Commande de l'utilisateur traitée en amont par le programme.
482     */
483    private void goRoom(Command pCommand) 
484    {
485        if(!pCommand.hasSecondWord()) {
486            // if there is no second word, we don't know where to go...
487            this.aGui.println("Go where? \n");
488            return;
489        }
490
491        String vDirection = pCommand.getSecondWord();
492
493        // Try to leave current room.
494        Room vNextRoom = this.aPlayer.getCurrentRoom().getExit(vDirection, this.aRooms.values(), this.aRooms.size());
495
496        if (vNextRoom == null)
497            this.aGui.println("There is no door!\n");
498        else {
499            this.aPlayer.addRoom(aPlayer.getCurrentRoom());
500            this.aPlayer.changeRoom(vNextRoom);
501            this.printLocationInfo();
502            if(this.aPlayer.getRoomImageName() != null)
503                this.aGui.showImage(this.aPlayer.getRoomImageName());
504        }
505        if( this.aPlayer.getCurrentRoom().getImageName().equals("Images/Final.jpg") ) {
506            this.aGui.println( this.aPlayer.finalFight() );
507            endGame();
508        }
509        this.aGui.update();
510    }
