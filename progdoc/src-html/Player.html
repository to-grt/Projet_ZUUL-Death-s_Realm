004/**
005 * Cette classe represente le joueur ainsi que ses differentes caractéristiques
006 */
007public class Player

009    private int aMaxWeight;

011    private String aName;

016    /**
017     * Constructeur de la classe Player.
018     * @param pMaxWeight    un int indiquant la capacité maximale du Player en kg.
019     * @param pName         un String contenant le nom du Player.
020     * @param pRoom         un Room contenant la Room actuelle du Player.
021     */
022    public Player(final int pMaxWeight, final String pName, final Room pRoom)
023    {
024        this.aMaxWeight = pMaxWeight;
025        this.aCurrentWeight = 0;
026        this.aName = pName;
027        this.aCarriedItems = new ItemList();
028        this.aCurrentRoom = pRoom;
029        this.aPreviousRooms = new Stack<Room>();
030    }

112    /**
113     * Permet de d'ajouter une Room dans l'historique des pièces du joueur (aPreviousRoom)
114     * @param pRoom une Room qui va être ajoutée dans l'attribut aPreviousRooms.
115     */
116    public void addRoom(final Room pRoom) {
117        this.aPreviousRooms.push(pRoom);
118    }

120    /**
121     * Permet de retirer une Room de l'historique des pièces du joueur (aPreviousRoom)

124    public Room removeRoom() {
125        return( this.aPreviousRooms.pop() );

128    /**
129     * Fonction permettant de prendre un item présent dans la salle.
130     * @param pItem un Item pouvant être prit par le Player.
131     * @return un String contenant sous un certain format quel Item à été prit.
132     */
133    public String pickItem(final Item pItem) {
134        this.aCarriedItems.addItem(pItem.getName(), pItem); 
135        this.setCurrentWeight(this.getCurrentWeight() + pItem.getPoids());
136        if( pItem.getName().equals("bag") ) { this.setMaxWeight(this.getMaxWeight() + 20); }
137        return("You just picked up a " + pItem.getName() + ". \n");
138    }
139    
140    /**
141     * Fonction permettant de lacher un item porté par le joueur.
142     * @param pItem un Item pouvant être déposé par le Player.
143     * @return un String contenant sous un certain format quel Item à été déposé.
144     */
145    public String dropItem(final Item pItem) {
146        this.aCarriedItems.removeItem(pItem.getName());
147        this.setCurrentWeight(this.getCurrentWeight() - pItem.getPoids());
148        if( pItem.getName().equals("bag") ) { this.setMaxWeight(this.getMaxWeight() - 20); }
149        return("You just dropped a " + pItem.getName() + ".");
150    }

196    /**
197     * Retourne l'Item voulu si il est présent dans la salle actuelle (sinon null)
198     * @param pName un Sring contenant le nom d'un Item (ou non).

200     * actuelle du Player, sinon renvoit null.
201     */
202    public Item getItemFromRoom(final String pName)
203    {
204        if(this.aCurrentRoom.getItem( pName ) == null ) return(null);
205        else return( this.aCurrentRoom.getItem( pName ) );
206    }

208    /**
209     * Retourne l'Item voulu si il est présent dans l'inventiare du joueur (sinon null)
210     * @param pName un Sring contenant le nom d'un Item (ou non).
211     * @return un Item si et seulement si le nom passé en paramètre correspond à un Item de la ItemList
212     * actuelle du Player, sinon renvoit null.
213     */
214    public Item getItemFromPlayer(final String pName)
215    {
216        if(this.aCarriedItems.getItem( pName ) == null ) return(null);
217        else return( this.aCarriedItems.getItem( pName ) );
218    }
219    
220    /**
221     * Retourne l'Item voulu (si il existe dans les items portés ou dans les items de la salle)
222     * @param pName un String contenant le nom de l'objet voulu (ou non).
223     * @return un Item si celui-ci existe dans la Room actuelle du Player ou dans l'ItemList du Player, sinon null.
224     */
225    public Item getItem(final String pName)
226    {
227        if( !isItemHere( pName ) ) return( null );
228        else {
229            Item vItem;
230            if( getItemFromPlayer( pName ) != null ) vItem = getItemFromPlayer( pName );
231            else vItem = getItemFromRoom( pName );
232            return(vItem);
233        }
234    }
235    
236    /**
237     * Enleve l'item d'un Player defintivement du jeu.
238     * @param pItem un Item allant être enlevé de l'inventaire d'un Player.
239     */
240    public void removeItem(final Item pItem)
241    {
242        this.setCurrentWeight( this.getCurrentWeight() - pItem.getPoids() );
243        this.aCarriedItems.removeItem(pItem.getName());
244    }

246    /**
247     * Fonction permettant d'obtenir sous forme de String une description de l'inventaire actuel du joueur
248     * @return un String contenant une liste formatée des Item de l'ItemList aCarriedItems du Player.
249     */
250    public String getPlayerInventory()
251    {
252        String vTotal = new String();
253        if( this.aCarriedItems.getTotalString() != "" ) {
254            vTotal += "this inventory contains : \n" + this.aCarriedItems.getTotalString() + "\n and the total weight of your inventory is : " + this.aCarriedItems.getTotalWeight() + " / " + this.aMaxWeight + "kg";
255        }
256        else vTotal = "this inventory is empty ! \n";
257        return( vTotal );
258    }

260    /**
261     * Fonction permettant de déterminer si un Item est présent dans la salle actuelle ou sur le joueur
262     * @param pName un String contenant le nom de l'Item cible.
263     * @return un boolean indiquant si l'Item dont le nom est donné en paramètre est dans la Room actuelle du Player
264     * ou dans l'ItemList aCarriedItem du Player.
265     */
266    public boolean isItemHere( final String pName )
267    {
268        if( getItemFromPlayer( pName ) == null && getItemFromRoom( pName ) == null) return false;
269        return true;
270    }

272    /**
273     * Fontion permettant de manger l'Item (si possible) passé en parametre
274     * @param pName un String contenant le nom l'Item cible de la commande eat().
275     * @return un String contenant la réponse de la fonction face à cette demande. Si l'Item n'existe pas, s'il
276     * n'est pas mangeable, si il à un effet spécial lorsque le joueur le mange ou lorsque il n'y a pas d'effet spécial.
277     */
278    public String eatItem( final String pName )
279    {
280        String vResult = new String();
281        
282        if( getItemFromPlayer( pName ) != null)
283        {
284            if( getItemFromPlayer( pName ).getEatable() )
285            {
286                Item vAteItem = getItemFromPlayer( pName );
287                if( vAteItem.getEffect().equals("+5kg"))
288                {
289                    setMaxWeight( getMaxWeight() + 5 );
290                    vResult += "you just gained 5kg on your max weight  \n";
291                }
292                removeItem(vAteItem);
293                return( "You just eated a " + vAteItem.getName() + ", you are not hungry anymore\n" + vResult );
294            }
295            return( "Humm not sure it is a good idea to eat that" );
296        }
297        return( "this item is currently not in your inventory or does not exist" );
298    }

300    /**
301     * Fonction décrivant la charge du beamer sur le Player
302     * @return un String contenant le réponse de la fonction par rapport à la demande du joueur.
303     */
304    public String chargeBeamer()
305    {
306        if( getItemFromPlayer("beamer") == null ) { return "You currently don't carry the beamer"; }
307        Beamer vBeamer = (Beamer)(getItemFromPlayer("beamer"));
308        vBeamer.setIsCharged(true);
309        vBeamer.setMemorizedRoom(this.aCurrentRoom);
310        return "The beamer has been charged";
311    }
312    
313    /**
314     * Fonction décrivant l'utilisation du beamer sur le Player.
315     * @return un String contenant la réponse de la focntion par rapport à la demande du joueur.
316     */
317    public String fireBeamer()
318    {
319        if( getItemFromPlayer("beamer") == null ) { return "You currently don't carry the beamer"; }
320        Beamer vBeamer = (Beamer)(getItemFromPlayer("beamer"));
321        if( vBeamer.getIsCharged() == false ) { return "The beamer is not charged ! You can't use it for now"; }
322        vBeamer.setIsCharged(false);
323        changeRoom( vBeamer.getMemorizedRoom() );
324        vBeamer.setMemorizedRoom(null);
325        return "It worked, you have been teleported successfully !";
326    }

328    /**
329     * Fonction pour le dernier combat du jeu.
330     * @return un String contenant le résultat du combat final.
331     */
332    public String finalFight()
333    {
334        int vTotalStrenght = 0;
335        int vTotalDefense = 0;
336        int vTotalMiss = 0;
337        for( Item pItem : this.aCarriedItems.getValues() ) {
338            vTotalStrenght += pItem.getStrenght();
339            vTotalDefense += pItem.getDefense();
340            vTotalMiss += pItem.getMiss();
341        }
342        double vScore = 0.5 * vTotalStrenght + 0.4 * vTotalDefense + 0.1 * vTotalMiss;
343        
344        if( vScore > 94.0 ) { return(win()); }
345        return(lose());
346    }
