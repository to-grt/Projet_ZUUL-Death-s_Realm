002/**
003 * Cette classe sert à définir les commandes considérées comme valides.
004 * 
005 * @author  Michael Kolling and David J. Barnes + D.Bureau + T.Gueuret
006 * @version 2008.03.30 + 2013.09.15 + 2019.02.20
007 */
008public class CommandWords
009{
010    // tableau constant qui contient tous les mots de commande valides
011    private static final String[] sValidCommands = {
012        "go", "quit", "help", "look", "eat", "back", "test", "drop", "take", "inventory", "charge", "fire"
013    };
014
015    /**
016     * Constructeur par defaut de la classe CommandWords
017     */
018    public CommandWords()
019    {
020        // rien a faire pour le moment...
021    } // CommandWords()
022
023    /**
024     * Affiche toutes les commandes valides
025     * @return un String contenant la liste des Commandes valides.
026     */
027    public String getCommandList()
028    {
029        String vString = new String("");
030        for(String command : sValidCommands)
031        {
032            vString += command + " ";
033        }
034        return(vString);
035    } //showAll()
036    
037    /**
038     * Verifie si une String donnee fait partie des commandes valides. 
039     * @param pString la String a tester
040     * @return true si pString est une comande valide, false sinon
041     */
042    public boolean isCommand( final String pString )
043    {
044        // pour chacune des commandes valides (du tableau constant)
045        for ( int i=0; i<sValidCommands.length; i++ ) {
046            // si elle est egale a pString
047            if ( sValidCommands[i].equals( pString ) )
048                return true;
049        } // for
050        // si nous arrivons la, c'est que la commande
051        // n'a pas ete trouvee dans le tableau
052        return false;
053    } // isCommand()
054} // CommandWords
